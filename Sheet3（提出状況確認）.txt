Option Explicit

Private Sub Worksheet_Change(ByVal target As Range)
    ' 提出状況管理シートのD列（提出日）が変更された場合のみ処理
    If Not Intersect(target, Me.Range("D:D")) Is Nothing Then
        If target.CountLarge = 1 Then ' 単一セル変更のみ対応
             If target.Row > MUNILIST_HEADER_ROW Then ' ヘッダー行より下
                ' 変更がD列で、かつ単一タスク表示モードの時のみ
                If target.Column = Me.Columns("D").Column And gSubmissionViewMode = VIEW_MODE_SINGLE_TASK Then
                    Call M_Submission.HandleSubmissionDateChange(target)
                End If
            End If
        End If
    End If
End Sub
